KNOWN_COMMAND_NAMES <- c(
  "alpha", "beta", "gamma", "delta", "epsilon", "zeta", "eta", "theta",
  "iota", "kappa", "lambda", "mu", "nu", "xi", "omicron", "pi", "rho",
  "sigma", "tau", "upsilon", "phi", "chi", "psi", "omega",
  "Gamma", "Delta", "Theta", "Lambda", "Xi", "Pi", "Sigma", "Upsilon",
  "Phi", "Psi", "Omega", "sin", "cos", "tan", "csc", "sec", "cot",
  "log", "ln", "exp", "sqrt", "frac", "sum", "int", "infty", "mathrm"
)

normalize_latex_math <- function(text) {
  if (is.null(text) || text == "") return("")
  
  # Strip % comments respecting \%
  text <- str_replace_all(text, "(?<!\\\\)%.*$", "")
  
  # Strip preamble/end document if present
  if (str_detect(text, fixed("\\begin{document}"))) {
    text <- str_split(text, fixed("\\begin{document}"), n = 2)[[1]][2]
  }
  if (str_detect(text, fixed("\\end{document}"))) {
    text <- str_split(text, fixed("\\end{document}"))[[1]][1]
  }
  
  # Standardize brace spacing
  text <- str_replace_all(text, "\\s*\\{\\s*", " { ")
  text <- str_replace_all(text, "\\s*\\}\\s*", " } ")
  text <- str_replace_all(text, "\\s*\\(", " (")
  text <- str_replace_all(text, "\\s+", " ")
  return(text)
}

get_letters_abbrs <- function(abbr_str) {
  pattern <- "(\\\\[a-zA-Z]+)|([A-Z][0-9]+)|([A-Z])([a-z]+)?|([a-z])|[0-9]+"
  matches <- str_extract_all(abbr_str, pattern)[[1]]
  
  repr_items <- c()
  for (m in matches) {
    if (str_starts(m, "\\\\")) {
      cmd_name <- str_sub(m, 2)
      if (cmd_name %in% KNOWN_COMMAND_NAMES) {
        repr_items <- c(repr_items, tolower(cmd_name))
      } else {
        repr_items <- c(repr_items, tolower(str_sub(cmd_name, 1, 1)))
      }
    } else {
      repr_items <- c(repr_items, tolower(str_sub(m, 1, 1)))
    }
  }
  return(repr_items)
}

get_letters_words <- function(word) {
  word <- str_trim(word)
  if (word == "") return("")
  
  word_clean <- str_replace_all(word, "^\\\\[a-zA-Z]+\\{?", "")
  word_clean <- str_replace_all(word_clean, "[\\$\\{\\}]", "")
  
  first_letter <- str_extract(word_clean, "[a-zA-Z0-9]")
  if (!is.na(first_letter)) return(tolower(first_letter))
  return("")
}

collect_abbreviations <- function(text) {
  pattern <- "((?:[\\w\\\\\\$\\{\\}]+[ -]+){1,10}(?:[\\w\\\\\\$\\{\\}]+)[ -]?)\\((?=[^\\(\\)\\,]*[A-Z])([^\\(\\)\\,]*[a-zA-Z0-9]{2,}[^\\(\\)\\,]*)\\)"
  
  matches <- str_match_all(text, pattern)[[1]]
  if (length(matches) == 0) {
    return(data.frame(Abbreviation=character(), Full_Phrase=character(), Usage=numeric(), Abbr_Match=numeric(), Word_Match=numeric()))
  }
  
  results <- list()
  for (i in 1:nrow(matches)) {
    phrase_part <- str_trim(matches[i, 2])
    abbr_part <- str_trim(matches[i, 3])
    
    clean_abbr <- str_replace_all(abbr_part, "[\\(\\)]", "")
    usage_pattern <- paste0("(?<![a-zA-Z\\(\\)])", regex_escape(clean_abbr), "(?![a-zA-Z\\)\\)])")
    usage_count <- tryCatch(length(str_extract_all(text, usage_pattern)[[1]]), error = function(e) 0)
    
    words_ahead <- str_split(phrase_part, "([ -]+)")[[1]]
    words_ahead <- words_ahead[words_ahead != ""]
    
    abbr_items <- get_letters_abbrs(abbr_part)
    word_comparables <- sapply(words_ahead, get_letters_words)
    
    match_indices <- rep(-1, length(abbr_items))
    last_word_idx <- length(words_ahead)
    
    if (length(abbr_items) > 0 && length(words_ahead) > 0) {
      for (a_idx in rev(seq_along(abbr_items))) {
        target <- abbr_items[a_idx]
        if (target == "") next
        
        for (w_idx in rev(seq_len(last_word_idx))) {
          if (word_comparables[w_idx] == "") next
          if (str_starts(word_comparables[w_idx], fixed(target))) {
            match_indices[a_idx] <- w_idx
            last_word_idx <- w_idx - 1
            break
          }
        }
      }
    }
    
    successful_matches <- match_indices[match_indices != -1]
    abbr_match_pct <- length(successful_matches) / length(abbr_items)
    
    word_match_pct <- 0.0
    full_phrase <- ""
    
    if (length(successful_matches) > 0) {
      min_idx <- min(successful_matches)
      phrase_slice <- words_ahead[min_idx:length(words_ahead)]
      full_phrase <- paste(phrase_slice, collapse = " ")
      
      full_phrase <- str_replace_all(full_phrase, "\\s+([\\-\\s])\\s+", "\\1")
      
      matchable_words <- sum(word_comparables[min_idx:length(word_comparables)] != "")
      if (matchable_words > 0) {
        word_match_pct <- length(successful_matches) / matchable_words
      }
    }
    
    results[[i]] <- data.frame(
      Abbreviation = abbr_part,
      Full_Phrase = full_phrase,
      Usage = usage_count,
      Abbr_Match = abbr_match_pct,
      Word_Match = word_match_pct,
      stringsAsFactors = FALSE
    )
  }
  
  return(do.call(rbind, results))
}

regex_escape <- function(string) {
  str_replace_all(string, "([\\.\\+\\*\\?\\^\\$\\(\\)\\[\\]\\{\\}\\|\\\\\\-])", "\\\\\\1")
}

format_exports <- function(df, type) {
  if (nrow(df) == 0) return("No items matching parameters to display.")
  
  if (type == "tabular") {
    rows <- paste0(df$Abbreviation, " & ", df$Full_Phrase, " \\\\")
    return(paste(c("\\begin{tabular}{ll}", "\\hline", "\\textbf{Abbreviation} & \\textbf{Full Name} \\\\", "\\hline", rows, "\\hline", "\\end{tabular}"), collapse="\n"))
  } else if (type == "nomenclature") {
    rows <- paste0("\\nomenclature{", df$Abbreviation, "}{", df$Full_Phrase, "}")
    return(paste(c("\\usepackage{nomencl}", "\\makenomenclature", rows), collapse="\n"))
  } else if (type == "footnote") {
    # Generate unified conditional Quarto output blocks
    latex_items <- paste0("\\\\textbf{", df$Abbreviation, "}: ", df$Full_Phrase)
    latex_str <- paste(latex_items, collapse = "; ")
    
    typst_items <- paste0("*_", df$Abbreviation, "_* ", df$Full_Phrase)
    typst_str <- paste(typst_items, collapse = "; #h(8pt) ")
    
    block <- paste0(
      '::: {.content-visible when-format="pdf"}\n',
      '```{=latex}\n',
      '\\begingroup\n',
      '\\renewcommand\\thefootnote{}\n',
      '\\footnotetext{\\textbf{Abbreviations:} ', latex_str, '.}\n',
      '\\endgroup\n',
      '```\n',
      ':::\n\n',
      '::: {.content-visible when-format="typst"}\n',
      '
```{=typst}\n',
      '#place(\n',
      '  bottom + left,\n',
      '  clearance: 0pt,\n',
      '  block(\n',
      '    width: 100%,\n',
      '    stroke: (top: 0.5pt + luma(150)),\n',
      '    inset: (top: 6pt),\n',
      '    [\n',
      '      #set text(size: 8.5pt)\n',
      '      *Abbreviations:* ', typst_str, '.\n',
      '    ]\n',
      '  )\n',
      ')\n',
      '```\n',
      ':::\n'
    )
    return(block)
  } else {
    return(paste0(df$Abbreviation, ": ", df$Full_Phrase, collapse = "; \n"))
  }
}

# =============================================================================
# 3. SERVER REACTIVE CONTROL HOOKS
# =============================================================================
server <- function(input, output, session) {
  
  raw_extracted_df <- reactive({
    req(input$file_upload)
    
    file_lines <- readLines(input$file_upload$datapath, warn = FALSE, encoding = "UTF-8")
    raw_text <- paste(file_lines, collapse = "\n")
    
    normalized <- normalize_latex_math(raw_text)
    collect_abbreviations(normalized)
  })
  
  processed_df <- reactive({
    df <- raw_extracted_df()
    if (nrow(df) == 0) return(df)
    
    if (input$dedup == "Yes" && "Abbreviation" %in% names(df)) {
      df <- df[!duplicated(df$Abbreviation), ]
    }
    
    df <- subset(df, Usage >= input$min_usage & 
                     Abbr_Match >= input$min_abbr & 
                     Word_Match >= input$min_word)
                     
    if (nrow(df) == 0) return(df)
    
    sort_mapping <- list(
      "Abbreviation" = list(col = "Abbreviation", dec = FALSE),
      "Usage" = list(col = "Usage", dec = TRUE),
      "% Abbr Matched" = list(col = "Abbr_Match", dec = TRUE),
      "% Words Matched" = list(col = "Word_Match", dec = TRUE)
    )
    
    cfg <- sort_mapping[[input$sort_col]]
    if (!is.null(cfg)) {
      df <- df[order(df[[cfg$col]], decreasing = cfg$dec), ]
    }
    
    return(df)
  })
  
  output$table_render <- renderTable({
    df <- processed_df()
    if (nrow(df) == 0) {
      return(data.frame(Status = "Upload a document file or adjust filters to reveal valid abbreviations."))
    }
    
    df$Abbr_Match <- sprintf("%.1f%%", df$Abbr_Match * 100)
    df$Word_Match <- sprintf("%.1f%%", df$Word_Match * 100)
    
    names(df) <- c("Abbreviation", "Full Phrase", "Usage", "% Abbr Matched", "% Words Matched")
    return(df)
  }, sanitize.text.function = function(x) x)
  
  output$export_render <- renderText({
    format_exports(processed_df(), input$export_fmt)
  })
}

shinyApp(ui, server)
